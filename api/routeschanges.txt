function createQRLink (link, width, height) {

  var svgStr = qrimage.imageSync(link, {type: 'svg'})

  var idx = svgStr.indexOf('path') - 2

  var fixedSvg = svgStr.substr(0, idx) + ' width="' + width + '" height="' + height + '"' + svgStr.substr(idx)



  return fixedSvg

}
